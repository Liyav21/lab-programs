public class sumofn{
    public static void main(String []args){
        int n=10,sum=0;
        for(int i=1;i<=n;i++){
            sum=sum+i;
        }
        System.out.println("The sum of first "+n+"  natural number is "+sum);
    }
    
 }



//OUTPUT: The sum of first 10  natural number is 55
